Functions

The first input to the functions is a dataset so the functions can be chained together in a sequence. The inputs to the functions are based on the toggles on the app.

Input Button Notes
adj_ch = TRUE/FALSE Apply 2024 Settings = ON/OFF Only for foursomes, greensomes, scramble when button = ON
diff90 = TRUE/FALSE Apply 2024 Settings = ON/OFF Only for 4BBB matchplay when button = ON
crp = TRUE/FALSE CH with CR-Par = ON/OFF Button is always ON if Apply 2024 Settings = ON. Otherwise can be either ON or OFF
round = TRUE/FALSE Round CH = ON/OFF Button defaults to OFF when Apply 2024 Setting = OFF, but can be switched ON after that without affecting any other buttons
rmatch = TRUE/FALSE Round Matchplay CH = ON/OFF Rounds CH for all matchplay formats irrespective of the Round CH button
dp = 0/1 Round Scramble TH = ON/OFF Final rounding used only for scramble formats
score = “high” None Fixed to use the maximum par to adjust CH


Course Handicap

The function course_handicap outputs all CH variables including the adjusted CH for both maximum par (chah) and minimum par (chal). Rounding is applied based on inputs round=TRUE (Round CH toggle) or rmatch=TRUE (Round matchplay CH toggle).
The functions round0 and round1 rounds the input to 0 or 1 decimal place.

Show code for function: course_handicap 
course_handicap <- function(data, 
                            index_var = "hi", 
                            holes = 18, 
                            round = TRUE, 
                            crp = FALSE,
                            hi_round = FALSE,
                            rmatch = FALSE) {

  data |>
    mutate(
      index := !!rlang::sym(index_var),
      index = round1(index),
      player = row_number(),
      crp = ifelse(holes == 9, TRUE, crp),
      minpar = min(par, na.rm=TRUE),
      maxpar = max(par, na.rm=TRUE),
      mincr = min(cr, na.rm=TRUE),
      maxcr = max(cr, na.rm=TRUE)
    ) |>
    rowwise() |>
    mutate(
      ch1 = case_when(
        holes == 18 & crp == FALSE ~ index * sr / 113,
        holes == 18 & crp == TRUE ~ index * sr / 113 + cr - par,
        holes == 9 & hi_round == FALSE ~ (index / 2) * sr / 113 + cr - par,
        holes == 9 & hi_round == TRUE ~ round1(index / 2) * sr / 113 + cr - par
      )
    ) |>
    mutate(
      ch1a = ch1 + par - minpar,
      ch1b = ch1 + par - maxpar,
      ch = ifelse(round == TRUE | rmatch == TRUE, round0(ch1), ch1),
      chah = ifelse(round == TRUE | rmatch == TRUE, round0(ch1b), ch1b),
      chal = ifelse(round == TRUE | rmatch == TRUE, round0(ch1a), ch1a)
      )
}

Playing Handicap

The function playing_handicap takes the output dataset from the course_handicap function and applies the allowances (the function add_format provides the required allowance for the selected format ). This function also calculates all the individual mixed tee adjustments using the crp input (CR-Par toggle) for both medal and stableford scoring formats.

Show code for function: playing_handicap 
playing_handicap <- function(data, 
                             allowance = 100, 
                             holes = 18, 
                             crp = FALSE) {
  data |>
    ungroup() |>
    mutate(ph = ch * allowance / 100) |>
    mutate(
      pts = case_when(
        holes == 18 & crp == FALSE ~ 36 - round0(cr - par),
        holes == 18 & crp == TRUE ~ 36,
        holes == 9 ~ 18
      ),
      stb = max(pts, na.rm = TRUE) - pts,
      medal = case_when(
        holes == 18 & crp == FALSE ~ cr - mincr,
        holes == 9 | crp == TRUE ~ par - minpar
      ),
       # note: common par and stb1 for use in foursomes/greensomes when crp=TRUE (existing calculation)
      commonpar = min(par, na.rm = TRUE),
      pts1 = 36 - round0(cr - commonpar),
      stb1 = max(pts1, na.rm = TRUE) - pts1,
      
      phm = round0(ph + medal),
      phs = round0(ph + stb),
      ph1 = ph,
      ph = round0(ph)
    ) |>
    filter(!is.na(ch))
}


4BBB Matchplay

The function matchplay takes the output from the course_handicap function. The playing_handicap function within the function is used to get phm and calculate s.
It uses the input diff90 (=TRUE if 2024setting toggle = ON) to apply 100% allowance to the CH and 90% allowance to the S. Otherwise it defaults to the current format 90% allowance and s = full difference to the lowest phm.

Show code for function: matchplay 
matchplay <- function(data, 
                      n = 2, 
                      holes = 18, 
                      crp = FALSE, 
                      diff90=FALSE) {
  data |>
    mutate(allowance = ifelse(n == 4 & diff90 == FALSE, 90, 100)) |> 
    playing_handicap(
      allowance = allowance,
      holes = holes,
      crp = crp
    ) |>
    ungroup() |>
    mutate(
      row_n = row_number(),
      group = ceiling(row_n / n)
    ) |>
    group_by(group) |>
    mutate(s = phm - min(phm, na.rm = TRUE)) |>
    ungroup() |>
    rowwise() |>
    mutate(s = ifelse(diff90 == TRUE, round0(s*0.9), s))
}

Foursomes & Greensomes

The function alternate_shot takes the output from the playing_handicap function and calculates the team handicaps using either the actual or adjusted CH (if 2024setting toggle = ON).
The input score is for using either the highest or the lowest par to adjust the CH. On the app, we’d always use the highest/maximum par so default score=high.

The function also calculates the combined CH (sumch) and difference in combined CH (dsumch) for use in ONLY the foursomes matchplay format when 2024setting toggle = ON.
Additional rounding to CH is applied based on the input rmatch (Round Matchplay CH toggle).

Show code for function: alternate_shot 
alternate_shot <- function(data,
                           holes = 18,
                           crp = FALSE,
                           type = "greensomes",
                           adj_ch = FALSE,
                           score = "high",
                           rmatch = FALSE) {

  weights <- tibble(
    type = rep(c("greensomes", "foursomes"), each = 2),
    rank = c(1, 2, 1, 2),
    weights = c(0.60, 0.4, 0.5, 0.5)
  )

  data |>
    ungroup() |>
    mutate(
      type = type,
      type = ifelse(str_detect(format, "Foursomes") == TRUE, "foursomes", "greensomes"),
      row_n = row_number(),
      group = ceiling(row_n / 2),
      mgroup = ceiling(row_n / 4)
    ) |>
    rowwise() |>
    mutate(
      cha1 = ifelse(score == "high", chah, chal),
      cha1 = ifelse(rmatch == TRUE & str_detect(format, "matchplay") == TRUE, round0(cha1), cha1),
      ch = ifelse(rmatch == TRUE & str_detect(format, "matchplay") == TRUE, round0(ch), ch)
      ) |>
    group_by(group) |>
    mutate(
      gsize = length(ch),
      rank = ifelse(gsize == 2, rank(ifelse(adj_ch == TRUE & crp == TRUE, cha1, ch), ties.method = "random"), NA)
    ) |>
    left_join(weights, by = c("type", "rank")) |>
    mutate(
      th = ifelse(adj_ch == TRUE & crp == TRUE, sum(cha1 * weights), sum(ch * weights)),
      adj_m = ifelse(adj_ch == TRUE  & crp == TRUE, 0, sum(medal / 2)),
      adj_s = ifelse(adj_ch == TRUE & crp == TRUE, 0, ifelse(holes == 9 | crp == TRUE, adj_m, sum(stb1 / 2))
      ),
      thm = ifelse(row_n / group == gsize, round0(th + adj_m), NA),
      ths = ifelse(row_n / group == gsize, round0(th + adj_s), NA),
      sumch1 = ifelse(str_detect(format, "Foursomes") == TRUE, 2*th, NA),
      sumch = ifelse(str_detect(format, "Foursomes") == TRUE, round0(2*(th+adj_m)), NA),
    ) |>
    group_by(mgroup) |>
    mutate(
      s = ifelse(!is.na(thm), thm - min(thm, na.rm = TRUE), NA),
      dsumch = ifelse(!is.na(sumch), sumch - min(sumch, na.rm = TRUE), NA),
      s1 = ifelse(!is.na(sumch) & !is.na(s), round0(0.5*(sumch - min(sumch, na.rm = TRUE))), NA)
      ) |>
    ungroup() |>
    select(-rank, -gsize, -mgroup)
}

Scramble

The function scramble takes the output from the course_handicap function applies the playing_handicap function. It uses either the actual CH or the adjusted CH (if 2024setting = ON) for the calculations. The input score is for using the highest or the lowest par to adjust the CH. On the app, we use only the highest/maximum par so the default input is fixed at score=high. The input dp applies the rounding at the end (Round scramble TH toggle on the app, with ON = 0dp, OFF = 1dp)

Show code for function: scramble 
scramble <- function(data,
                     n = 2,
                     holes = 18,
                     crp = FALSE,
                     adj_ch=FALSE,
                     score="high",
                     dp = 0) {

  weights <- tibble(
    size = c(2, 2, 3, 3, 3, 4, 4, 4, 4),
    rank = c(1, 2, 1, 2, 3, 1, 2, 3, 4),
    weights = c(0.35, 0.15, 0.30, 0.2, 0.1, 0.25, 0.2, 0.15, 0.1)
  )

  data |>
    ungroup() |>
    mutate(
      tees = ifelse(length(unique(cr)) > 1, "mixed", "single"),
      across(c(cr, par), ~ ifelse(tees == "mixed" & n == 3 &
                                    nrow(data) == 4 & player == 4, NA, .)),
      across(c(cr, par), ~ ifelse(tees == "mixed" & n == 2 &
                                    nrow(data) == 3 & player == 3, NA, .))
           ) |>
    playing_handicap(holes = holes, crp = crp) |>
    ungroup() |>
    mutate(
      group = ceiling(row_number() / n)
    ) |>
    rowwise() |>
    mutate(cha1 = ifelse(score == "high", chah, chal)) |>
    group_by(group) |>
    mutate(
      row_n = row_number(),
      size = length(ch),
      rank = rank(ifelse(adj_ch == TRUE & crp == TRUE, cha1, ch), ties.method = "random")

    ) |>
    left_join(weights, by = c("size", "rank")) |>
    mutate(weights = ifelse(size < n, NA, weights)) |> 
    # show values only against the last player on a team
    mutate(
      adj = ifelse(row_n == size & !is.na(weights), 
                   ifelse(adj_ch == TRUE & crp == TRUE, 0, sum(medal / size)), NA),
      th = ifelse(row_n == size, ifelse(adj_ch == TRUE & crp == TRUE, sum(cha1 * weights),
                                        sum(ch * weights)), NA),
      thm = ifelse(row_n == size, ifelse(adj_ch == TRUE & crp == TRUE, sum(cha1 * weights),
                                         sum(ch * weights) + adj), NA)
    ) |> 
    mutate(
      thm = case_when(row_n == size & dp == 1 ~ round1(thm),
                     row_n == size & dp == 0 ~ round0(thm),
                     TRUE ~ NA)
    ) |>
    ungroup() |>
    filter(!is.na(ch)) |>
    select(-rank, -size, -row_n) 
    
}
